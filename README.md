2. The Database (Supabase)

Create a new project on Supabase.

Go to the SQL Editor and run the following schema to create the tables:

-- Create Clients Table
create table clients (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  name text not null,
  author text,
  version text,
  type text, -- 'Utility', 'CPvP', 'Anarchy'
  os text, -- 'Windows', 'Android', 'iOS'
  status text, -- 'Free', 'Paid', 'Private', 'Archive'
  description text,
  features text[], -- Array e.g. ['Fly', 'Killaura']
  download_url text, -- Use relative path: './artifacts/file.dll'
  config_url text,   -- Use relative path: './artifacts/configs/file.txt'
  verified boolean default false,
  death_date text,
  death_note text
);

-- Create Rankings Table
create table meta_rankings (
  id bigint generated by default as identity primary key,
  client_id bigint references clients(id),
  category text, -- 'CPVP' or 'SPVP'
  rank int,
  note text
);
